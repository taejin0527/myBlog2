def boyer_moore(pattern, text):
result = 0
pivot = len(pattern)-1
skip = list(pattern)
skip.reverse()

    while result!=1 and pivot<len(text):
        if pattern[len(pattern) - 1] != text[pivot]:
            if text[pivot] in skip:
                nums = skip.index(text[pivot])
                pivot += nums
            else:
                pivot += len(pattern)
        else:
            for i, j in zip(range(pivot, pivot - len(pattern), -1), skip):
                if text[i] == j:
                    result = 1
                else:
                    result = 0
                    pivot = i
                    break
    return result
