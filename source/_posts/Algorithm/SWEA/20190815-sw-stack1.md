def count_parenthesis(line):
stack = []

    for key in line:
        if key == "(" or key == "{":
            stack += key
        elif key == ")" or key == "}":
            if len(stack) == 0:
                stack += key
                break
            elif key == ")" and stack[-1] == "(":
                stack.pop()
            elif key == "}" and stack[-1] == "{":
                stack.pop()
            else:
                stack += key
                break
    isPare = 0
    if len(stack) == 0:
        isPare = 1

    return isPare
